import io
import os
import pandas as pd
import numpy as np
import statsmodels.api as sm
from statsmodels.stats.outliers_influence import variance_inflation_factor
from shiny import App, render, ui, reactive
from shinywidgets import output_widget, render_widget
import plotly.express as px
import plotly.graph_objects as go

        ui.input_select("x_col", "X Predictor", choices=[]),

        ui.input_numeric("sigma_val", "Outlier Sigma", value=3.0, step=0.1),

def server(input, output, session):
    current_page = reactive.Value(0)

    @reactive.calc
    def data():
        f = input.file1()
        if f: return pd.read_csv(f["datapath"])
        local_path = os.path.join(os.path.dirname(__file__), "app/data", "test.csv")
        if os.path.exists(local_path): return pd.read_csv(local_path)
        return pd.DataFrame()

    @reactive.effect
    def _update_cols():
        df = data()
        if not df.empty:
            cols = df.select_dtypes(include=[np.number]).columns.tolist()
            ui.update_select("x_col", choices=cols, selected=cols if cols else None)
            ui.update_select("y_col", choices=cols, selected=cols if len(cols) > 1 else None)

    @reactive.calc
    def model_results():
        df = data()
        if df.empty or not input.x_col() or not input.y_col(): return None
        df_reg = df[[input.x_col(), input.y_col()]].apply(pd.to_numeric, errors='coerce').dropna()
        X = sm.add_constant(df_reg[input.x_col()])
        model = sm.OLS(df_reg[input.y_col()], X).fit()
        influence = model.get_influence()
        outliers = df_reg[np.abs(influence.resid_studentized_internal) > input.sigma_val()]
        return {"model": model, "df": df_reg, "X": X, "infl": influence, "outliers": outliers}

    @render.text
    def equation():
        res = model_results()
        if not res: return "Awaiting Data..."
        p = res['model'].params
        return f"Equation: {input.y_col()} = ({p.iloc[1]:.4f} * {input.x_col()}) + ({p.iloc[0]:.4f})"

    @render_widget
    def plot_fit_interactive():
        res = model_results()
        if not res: return go.Figure()
        x_rng = np.linspace(res["df"][input.x_col()].min(), res["df"][input.x_col()].max(), 100)
        preds = res["model"].get_prediction(sm.add_constant(x_rng)).summary_frame(alpha=1-(input.conf_level()/100))
        fig = px.scatter(res["df"], x=input.x_col(), y=input.y_col(), title="Regression Fit with CI/PI & Outliers")
        fig.add_trace(go.Scatter(x=np.concatenate([x_rng, x_rng[::-1]]), 
                                 y=np.concatenate([preds['obs_ci_upper'], preds['obs_ci_lower'][::-1]]),
                                 fill='toself', fillcolor='rgba(150,150,150,0.2)', line_color='rgba(255,255,255,0)', name='95% PI'))
        fig.add_trace(go.Scatter(x=np.concatenate([x_rng, x_rng[::-1]]), 
                                 y=np.concatenate([preds['mean_ci_upper'], preds['mean_ci_lower'][::-1]]),
                                 fill='toself', fillcolor='rgba(0,100,255,0.2)', line_color='rgba(255,255,255,0)', name='95% CI'))
        fig.add_trace(go.Scatter(x=x_rng, y=preds['mean'], mode='lines', name='OLS Fit', line=dict(color='black')))
        if not res["outliers"].empty:
            fig.add_trace(go.Scatter(x=res["outliers"][input.x_col()], y=res["outliers"][input.y_col()], mode='markers+text',
                                     text=[f"Obs {i}" for i in res["outliers"].index], textposition="top center",
                                     marker=dict(color='red', size=10, symbol='circle-open'), name='Outliers'))
        fig.update_layout(template="simple_white", font_family="Times New Roman", xaxis_title=input.x_col(), yaxis_title=input.y_col())
        return fig

    @render_widget
    def plot_hist():
        res = model_results()
        if not res: return go.Figure()
        fig = px.histogram(res["model"].resid, title="Error Distribution", labels={'value': 'Residuals', 'count': 'Frequency'})
        fig.update_traces(marker_line_color='white', marker_line_width=1)
        fig.update_layout(xaxis_title="Residuals", yaxis_title="Frequency")
        return fig

    @render_widget
    def plot_cooks():
        res = model_results()
        if not res: return go.Figure()
        fig = go.Figure(data=[go.Bar(y=res["infl"].cooks_distance[0], name="Cook's D")])
        fig.update_layout(title="Cook's Distance", xaxis_title="Observation Index", yaxis_title="Influence", 
                          yaxis=dict(nticks=10), template="simple_white")
        return fig

    @render.table
    def table_summary_1():
        res = model_results()
        if not res: return None
        df = res["model"].summary2().tables[1].reset_index().rename(columns={"index": "Variable"})
        
        def add_stars(p):
            if p < 0.001: return "***"
            if p < 0.01: return "**"
            if p < 0.05: return "*"
            if p < 0.1: return "."
            return ""
        
        df["Sig"] = df["P>|t|"].apply(add_stars)
        return df

    @render.table
    def table_summary_0(): return model_results()["model"].summary2().tables[0] if model_results() else None
    @render.table
    def table_summary_2(): return model_results()["model"].summary2().tables[2] if model_results() else None

    @render_widget
    def plot_heatmap():
        df = data()
        if df.empty: return go.Figure()
        corr = df.select_dtypes(include=[np.number]).corr()
        fig = px.imshow(corr, text_auto=".2f", color_continuous_scale='RdBu_r', title="Correlation Heatmap")
        fig.update_layout(width=700, height=700, xaxis_title="Variables", yaxis_title="Variables")
        return fig

    @render_widget
    def plot_resid_pred():
        res = model_results()
        if not res: return go.Figure()
        fig = px.scatter(x=res["df"][input.x_col()], y=res["model"].resid, title="Residuals vs Predictor", labels={'x': input.x_col(), 'y': 'Residuals'})
        fig.add_hline(y=0, line_dash="dash", line_color="red")
        fig.update_layout(xaxis_title=input.x_col(), yaxis_title="Residuals")
        return fig

    @render_widget
    def plot_resid_fit():
        res = model_results()
        if not res: return go.Figure()
        fig = px.scatter(x=res["model"].fittedvalues, y=res["model"].resid, title="Residuals vs Fitted", labels={'x': 'Fitted Values', 'y': 'Residuals'})
        fig.add_hline(y=0, line_dash="dash", line_color="red")
        fig.update_layout(xaxis_title="Fitted Values", yaxis_title="Residuals")
        return fig

    @render_widget
    def plot_resid_leverage():
        res = model_results()
        if not res: return go.Figure()
        fig = px.scatter(x=res["infl"].hat_matrix_diag, y=res["infl"].resid_studentized_internal, 
                          title="Residuals vs Leverage", labels={'x': 'Leverage', 'y': 'Std Residuals'})
        fig.update_layout(xaxis_title="Leverage", yaxis_title="Standardized Residuals")
        return fig

    @render_widget
    def plot_scale_location():
        res = model_results()
        if not res: return go.Figure()
        fig = px.scatter(x=res["model"].fittedvalues, y=np.sqrt(np.abs(res["infl"].resid_studentized_internal)), 
                          title="Scale-Location", labels={'x': 'Fitted Values', 'y': 'sqrt|Std Residuals|'})
        fig.update_layout(xaxis_title="Fitted Values", yaxis_title="√|Standardized Residuals|")
        return fig

    @render_widget
    def plot_qq():
        res = model_results()
        if not res: return go.Figure()
        from statsmodels.graphics.gofplots import ProbPlot
        qq = ProbPlot(res["model"].resid)
        fig = go.Figure(data=[go.Scatter(x=qq.theoretical_quantiles, y=qq.sample_quantiles, mode='markers', name='Residuals')])
        fig.add_trace(go.Scatter(x=[-3,3], y=[-3,3], mode='lines', line=dict(color='red', dash='dash'), name='Normal'))
        fig.update_layout(title="Normal Q-Q", xaxis_title="Theoretical Quantiles", yaxis_title="Sample Quantiles", template="simple_white")
        return fig

    @render.table
    def table_metrics():
        res = model_results()
        if not res: return None
        r = res['model'].resid
        return pd.DataFrame({"Metric": ["MSE", "RMSE", "MAE"], "Value": [f"{np.mean(r**2):.4f}", f"{np.sqrt(np.mean(r**2)):.4f}", f"{np.mean(np.abs(r)):.4f}"]})

    @render.table
    def table_vif():
        res = model_results()
        if not res: return None
        X_v = res["X"].values
        return pd.DataFrame({"Variable": res["X"].columns, "VIF": [f"{variance_inflation_factor(X_v, i):.4f}" for i in range(X_v.shape[1])]})

    @render.table
    def table_intervals():
        res = model_results()
        if not res: return None
        p = res["model"].get_prediction(res["X"]).summary_frame(alpha=1-(input.conf_level()/100))
        full = pd.concat([res["df"][[input.x_col()]], p], axis=1).reset_index(drop=True)
        start = current_page.get() * 10
        return full.iloc[start : start + 10]

    @reactive.effect
    @reactive.event(input.next_10)
    def _next(): current_page.set(current_page.get() + 1)
    @reactive.effect
    @reactive.event(input.prev_10)
    def _prev(): current_page.set(max(0, current_page.get() - 1))

    @render.download(filename="Report.pdf")
    def download_pdf(): yield b"PDF"

app = App(app_ui, server)
